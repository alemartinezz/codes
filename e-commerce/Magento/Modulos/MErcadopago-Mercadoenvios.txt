Add repository to your Magento installation composer.json file

"repositories": [ { "type": "vcs", "url": "https://github.com/mercadopago/cart-magento2" } ]

composer require mercadopago/magento2-plugin

bin/magento module:enable MercadoPago_Core

bin/magento setup:upgrade

/opt/lampp/bin/php bin/magento module:disable MercadoPago_MercadoEnvios
